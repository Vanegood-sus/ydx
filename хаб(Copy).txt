local touchStartPos, frameStartPos
local isDragging = false

imageFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
        isDragging = true
        touchStartPos = input.Position
        frameStartPos = imageFrame.Position
        TweenService:Create(imageFrame, TweenInfo.new(0.1), {Size = UDim2.new(0, imageSize-5, 0, imageSize-5)}):Play()
    end
end)

imageFrame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
        isDragging = false
        TweenService:Create(imageFrame, TweenInfo.new(0.1), {Size = UDim2.new(0, imageSize, 0, imageSize)}):Play()
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if isDragging and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement) then
        local delta = input.Position - touchStartPos
        imageFrame.Position = UDim2.new(
            frameStartPos.X.Scale,
            frameStartPos.X.Offset + delta.X,
            frameStartPos.Y.Scale,
            frameStartPos.Y.Offset + delta.Y
        )
    end
end)

local function loadPlayerAvatar()
    local userId = LocalPlayer.UserId
    local thumbType = Enum.ThumbnailType.AvatarThumbnail
    local thumbSize = Enum.ThumbnailSize.Size100x100
    local content, isReady = Players:GetUserThumbnailAsync(userId, thumbType, thumbSize)
    Avatar.Image = content
end

local function switchTab(tab)
    MenuTab.BackgroundColor3 = Color3.fromRGB(35, 35, 40)
    ScriptsTab.BackgroundColor3 = Color3.fromRGB(35, 35, 40)
    GamesTab.BackgroundColor3 = Color3.fromRGB(35, 35, 40)
    TrollTab.BackgroundColor3 = Color3.fromRGB(35, 35, 40)
    
    MenuTab.TextColor3 = Color3.fromRGB(180, 180, 180)
    ScriptsTab.TextColor3 = Color3.fromRGB(180, 180, 180)
    GamesTab.TextColor3 = Color3.fromRGB(180, 180, 180)
    TrollTab.TextColor3 = Color3.fromRGB(180, 180, 180)
    
    MenuFrame.Visible = false
    ScriptsFrame.Visible = false
    GamesFrame.Visible = false
    TrollFrame.Visible = false
    
    if tab == "menu" then
        MenuTab.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
        MenuTab.TextColor3 = Color3.fromRGB(255, 165, 50)
        TweenService:Create(ActiveTabIndicator, TweenInfo.new(0.2), {Position = UDim2.new(0, 0, 1, -3), Size = UDim2.new(0.25, 0, 0, 3)}):Play()
        MenuFrame.Visible = true
    elseif tab == "scripts" then
        ScriptsTab.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
        ScriptsTab.TextColor3 = Color3.fromRGB(255, 165, 50)
        TweenService:Create(ActiveTabIndicator, TweenInfo.new(0.2), {Position = UDim2.new(0.25, 0, 1, -3), Size = UDim2.new(0.25, 0, 0, 3)}):Play()
        ScriptsFrame.Visible = true
    elseif tab == "games" then
        GamesTab.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
        GamesTab.TextColor3 = Color3.fromRGB(255, 165, 50)
        TweenService:Create(ActiveTabIndicator, TweenInfo.new(0.2), {Position = UDim2.new(0.5, 0, 1, -3), Size = UDim2.new(0.25, 0, 0, 3)}):Play()
        GamesFrame.Visible = true
    else
        TrollTab.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
        TrollTab.TextColor3 = Color3.fromRGB(255, 165, 50)
        TweenService:Create(ActiveTabIndicator, TweenInfo.new(0.2), {Position = UDim2.new(0.75, 0, 1, -3), Size = UDim2.new(0.25, 0, 0, 3)}):Play()
        TrollFrame.Visible = true
    end
end

local function setupTabHoverEffects(tab)
    tab.MouseEnter:Connect(function()
        if tab.TextColor3 ~= Color3.fromRGB(255, 165, 50) then
            TweenService:Create(tab, TweenInfo.new(0.2), {TextColor3 = Color3.fromRGB(220, 220, 220)}):Play()
        end
    end)
    
    tab.MouseLeave:Connect(function()
        if tab.TextColor3 ~= Color3.fromRGB(255, 165, 50) then
            TweenService:Create(tab, TweenInfo.new(0.2), {TextColor3 = Color3.fromRGB(180, 180, 180)}):Play()
        end
    end)
end
